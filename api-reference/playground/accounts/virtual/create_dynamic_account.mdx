---
title: "Dynamic Account"
api: "POST /account/create/dynamic"
---

<ParamField header="requestId" type="string" required />
<ParamField body="name" type="string" required>
	Name to be displayed to the customer upon payment via his/her bank app.
</ParamField>
<ParamField body="amount" type="string">
	The maximum amount in which is expected to be collected via the account.
</ParamField>
<ParamField body="expTime" type="string" default="1800">
	The expiry time in seconds, in which the dynamic virtual account is to be active for
</ParamField>
<ParamField body="expDate" type="string">
	The expiry date in which the dynamic virtual account is to be active for
</ParamField>

<ResponseExample>

```json
{
	"statusCode": 00,
	"message": "Account successfully created",
	"data": {
		"accountNumber": "1951145523",
		"accountName": "Tony Stark",
		"reversalId": "res1324323"
	},
	"errors": []
}
```

</ResponseExample>
