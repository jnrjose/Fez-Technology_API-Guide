---
title: "Initiate Transfer"
api: "POST /transfer/initiate"
description: "This allows you to initiate a transfer request on behalf of your customer."
---

<Info>
	**Note**
	<br /> You can initiate transfer requests only with account numbers generated specifically
	for your customer.
</Info>

<ParamField header="requestId" type="string" required />
<ParamField body="nameEnquirySessionId" type="string" required></ParamField>
<ParamField body="sourceAccountName" type="string" required></ParamField>
<ParamField body="sourceAccountNumber" type="string" required></ParamField>
<ParamField body="beneficiaryAccountName" type="string" required></ParamField>
<ParamField body="beneficiaryAccountNumber" type="string" required></ParamField>
<ParamField body="beneficiaryBankCode" type="string" required></ParamField>
<ParamField body="amount" type="double" required></ParamField>
<ParamField body="narration" type="string" optional></ParamField>
